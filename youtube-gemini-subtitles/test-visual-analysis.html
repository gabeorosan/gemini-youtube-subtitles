The extension now captures video frames using the following process:

// Frame capture process async function captureVideoFrames(video) { const frames = []; const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); canvas.width = video.videoWidth || 640; canvas.height = video.videoHeight || 360; const duration = video.duration; const frameInterval = Math.max(5, duration / 10); for (let time = 0; time < duration; time += frameInterval) { video.currentTime = time; await waitForSeek(); ctx.drawImage(video, 0, 0, canvas.width, canvas.height); const frameData = canvas.toDataURL('image/jpeg', 0.8); frames.push({ timestamp: time, data: frameData }); } return frames; }
